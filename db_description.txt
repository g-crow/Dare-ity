-- Table: public.client

-- DROP TABLE public.client;

CREATE TABLE public.client
(
  id integer NOT NULL DEFAULT nextval('client_id_seq'::regclass),
  name character varying(80) NOT NULL,
  hash_password character varying(80) NOT NULL,
  is_npo boolean NOT NULL DEFAULT false,
  CONSTRAINT "Client Primary Key" PRIMARY KEY (id),
  CONSTRAINT "unique name" UNIQUE (name)
)
WITH (
  OIDS=FALSE
);
ALTER TABLE public.client
  OWNER TO dareity;





-- Table: public.client_dare

-- DROP TABLE public.client_dare;

CREATE TABLE public.client_dare
(
  client_dare_id integer NOT NULL DEFAULT nextval('client_dare_client_dare_id_seq'::regclass),
  client_id integer NOT NULL,
  dare_id integer NOT NULL,
  pledge_total money NOT NULL,
  npo_id integer NOT NULL,
  CONSTRAINT "primary client dare key" PRIMARY KEY (client_dare_id)
)
WITH (
  OIDS=FALSE
);
ALTER TABLE public.client_dare
  OWNER TO dareity;





-- Table: public.dare

-- DROP TABLE public.dare;

CREATE TABLE public.dare
(
  id_dare integer NOT NULL DEFAULT nextval('dare_id_dare_seq'::regclass),
  dare_title character varying(150) NOT NULL,
  dare_description text NOT NULL,
  npo_creator integer NOT NULL,
  dare_expires date NOT NULL,
  total_pledge_dollar_amount money,
  CONSTRAINT dare_primary_key PRIMARY KEY (id_dare),
  CONSTRAINT "unique dare title" UNIQUE (dare_title)
)
WITH (
  OIDS=FALSE
);
ALTER TABLE public.dare
  OWNER TO dareity;





-- Table: public.pledge

-- DROP TABLE public.pledge;

CREATE TABLE public.pledge
(
  id_pledge integer NOT NULL DEFAULT nextval('id_pledge_id_pledge_seq'::regclass),
  broadcaster_id integer NOT NULL,
  npo_id integer NOT NULL,
  dare_id integer NOT NULL,
  pledge_amount money NOT NULL,
  fulfilled boolean NOT NULL DEFAULT false,
  client_dare_id integer NOT NULL,
  CONSTRAINT "primary pledge key" PRIMARY KEY (id_pledge)
)
WITH (
  OIDS=FALSE
);
ALTER TABLE public.pledge
  OWNER TO dareity;
